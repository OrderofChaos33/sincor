1) Ideal Customer Profile (ICP) — crisp, machine-checkable

Company filters

Headcount: 20–1,000 (sweet spot: 50–300)

Industry: multi-site services, retail, F&B, DTC, SaaS, healthcare clinics, light manufacturing

Geography: US/Canada to start; expand per performance

Growth proxies (≥1 true): hiring velocity ↑, ad spend ↑, funding ≤ 36 months, >2 new locations in 18 months

Decision intent (≥1 signal): “market entry”, “new location”, “competitive analysis”, “pricing study”, RFPs for “research/BI/strategy”

People filters (must hit at least one title)

CEO, COO, CFO, VP/Director Strategy, Corp Dev, Business Ops, RevOps, Product Strategy

Pain signals (score boosters)

Recent PR/press launches; job posts: “market research”, “competitive intelligence”, “analyst”

High CPC keywords in their niche (expensive hesitations → value BI)

Seasonality approaching (HVAC summer/winter, retail holidays)

Review turbulence (ratings volatility last 90d)

2) Signals & weights (can tune live)
signals:
  size_midmarket:            +2.0  # 50–300 FTE
  growth_hiring:             +1.5  # ≥3 open roles in strategy/ops/sales
  funding_recent:            +1.2  # seed/series in last 36 months
  expansion_locations:       +2.0  # ≥2 new sites in 18 months
  intent_keywords:           +2.5  # “market entry”, “competitive analysis”, “RFP research”

  enterprise_slowcycle:      -0.8  # >1000 FTE unless strong intent

  MQL: 5.0

from dataclasses import dataclass
from typing import List, Dict, Any
import math, re

@dataclass
class Company:
    name: str
    domain: str
    headcount: int
    locations_18m: int
    city: str
    country: str
    funding_months_ago: int|None
    titles: List[str]
    job_titles: List[str]
    press_hits: List[str]
    review_volatility_90d: float
    sector_cpc_high: bool
    intent_text: str
    prior_research_vendor_mentions: bool

def sigmoid(x): return 1 / (1 + math.exp(-x))

class Scorer:
    def __init__(self, cfg: Dict[str, Any]): self.cfg = cfg

    def score(self, c: Company) -> Dict[str, float]:
        w, p = self.cfg["weights"], self.cfg["penalties"]
        s = 0.0

        # positive signals
        if self.cfg["headcount"]["sweet_min"] <= c.headcount <= self.cfg["headcount"]["sweet_max"]:
            s += w["size_midmarket"]
        if c.locations_18m >= 2:                  s += w["expansion_locations"]
        if c.funding_months_ago is not None and c.funding_months_ago <= 36:
            s += w["funding_recent"]
        if any(t in set(c.titles) for t in self.cfg["titles"]):
            s += w["exec_title_present"]
        if re.search(r"(market entry|competitive analysis|new location|pricing study|research RFP|business intelligence report)", c.intent_text, re.I):
            s += w["intent_keywords"]
        if c.review_volatility_90d >= 0.25:       s += w["review_volatility"]
        if c.sector_cpc_high:                     s += w["high_cpc_sector"]
        if c.prior_research_vendor_mentions:      s += w["prior_research_spend"]

        # penalties
        if c.headcount < self.cfg["headcount"]["min"]: s += p["headcount_too_small"]
        if c.headcount > self.cfg["headcount"]["max"]: s += p["enterprise_slowcycle"]

        fit = sigmoid(s/4.0)           # 0..1
        urgency = sigmoid((c.locations_18m + (1 if c.funding_months_ago and c.funding_months_ago<=12 else 0))/2.0)
        total = 0.6*fit + 0.4*urgency
        return {"fit": round(fit,3), "urgency": round(urgency,3), "total": round(total,3)}

def pitch(company: Company, score: Dict[str,float], cfg: Dict[str,Any]) -> str:
    price = cfg["pricing_anchor"]
    proof = []
    if company.locations_18m>=2: proof.append(f"{company.locations_18m} new locations in 18 months")
    if company.funding_months_ago and company.funding_months_ago<=36: proof.append("recent funding")
    if re.search(r"competitive|market entry|pricing", company.intent_text, re.I): proof.append("active strategy intent")
    proof_txt = "; ".join(proof) or "growth activity detected"

    return (
      f"Subject: Same intelligence, today — {company.name}\n\n"
      f"{company.name} can skip a $ {price['legacy_price']:,} / {price['legacy_timeline_days']}-day study.\n"
      f"SINCOR delivers equivalent decision intel in ~{price['sincor_timeline_hours']} hours for ${price['sincor_price']:,}.\n"
      f"Why you: {proof_txt}. Headcount ~{company.headcount}.\n\n"
      "Want a 2-page preview tailored to your market by tomorrow?\n"
      "— Reply 'PREVIEW' or hit: action://demo.start\n"
    )

7) Agent prompts (short, enforceable)

Finder — system

Collect companies with headcount 20–1,000 in US/CA. Prefer 50–300. Boost those showing expansion intent (keywords), recent funding (≤36mo), or 2+ new locations in 18mo. Return domains and evidence URLs. Avoid agencies/consultancies.

Enricher — system

For each domain, fetch titles (CEO/COO/CFO/VP Strategy/Corp Dev/RevOps), open roles mentioning research/intelligence/analyst, funding month deltas, locations opened in 18 months, review volatility, sector CPC. Normalize to the schema.

Scorer — system

Apply weights from config. Compute fit, urgency, total ∈ [0,1]. Attach per-signal rationale. Do not invent data. If unknown, leave null and do not award points.

Verifier — system

Check all claims against evidence URLs. If a claim lacks a source, remove it. Ensure pricing/timeline statements match anchors. If compliance risk, fail closed.

Pitcher — system

Generate ≤120-word email + ≤220-char LinkedIn DM. Include the pricing/timeline contrast and 1–2 evidence bullets. Close with one action: “PREVIEW” or link. No adjectives that imply guaranteed results.

8) Compliance anchors (lock in the flyer promise)

Legacy comparator: $25,000 / 56 days; SINCOR: $7,500 / ~4 hours (only state if true for that product tier; otherwise “hours, not months”).

Always pair assertions with “equivalent decision intel” / “same depth targets” framing, not “identical report by Firm X.”

Auto-insert evidence URLs for every rationale line.

(Your uploaded campaign doc matches this positioning; keep the “Same Intelligence. Today.” line available as a headline.

10) Sanity checks (catch “doesn’t function right” issues)

Idempotency: hash(domain) for dedupe; don’t re-email within 14 days

Null-safety: unknown fields must not award points

Explainability: include per-signal rationale in logs

Backtests: run the scorer on closed-won vs closed-lost to update weights weekly

Kill-switch: if reply rate < baseline for 3 days, pause outreach and surface top 5 rationales for human review (you)

from pydantic import BaseModel, Field, HttpUrl
from typing import List, Optional, Dict, Any
import math, re, hashlib, yaml, os

# Load config once
CONFIG_PATH = os.getenv("SCOUT_CONFIG", os.path.join(os.path.dirname(__file__), "..", "config.yaml"))
with open(CONFIG_PATH, "r") as f:

def sigmoid(x: float) -> float:
    return 1 / (1 + math.exp(-x))

class Evidence(BaseModel):

    url: Optional[str] = None

class CompanyIn(BaseModel):
    name: str
    domain: str
    headcount: Optional[int] = None

class ScoreOut(BaseModel):

    urgency: float
    total: float

class ScoredCompany(BaseModel):
    company: CompanyIn
    scores: ScoreOut

class PitchOut(BaseModel):
    subject: str
    email: str
    dm: str

def score_company(c: CompanyIn, cfg: Dict[str, Any]) -> ScoreOut:
    w, p = cfg["weights"], cfg["penalties"]
    rationale: Dict[str, float] = {}
    s = 0.0
    # positives
    if c.headcount is not None and cfg["headcount"]["sweet_min"] <= c.headcount <= cfg["headcount"]["sweet_max"]:
        s += w["size_midmarket"]; rationale["size_midmarket"] = w["size_midmarket"]

        s += w["expansion_locations"]; rationale["expansion_locations"] = w["expansion_locations"]
    if c.funding_months_ago is not None and c.funding_months_ago <= 36:
        s += w["funding_recent"]; rationale["funding_recent"] = w["funding_recent"]
    if any(t in set(c.titles) for t in cfg["titles"]):
        s += w["exec_title_present"]; rationale["exec_title_present"] = w["exec_title_present"]

        s += w["intent_keywords"]; rationale["intent_keywords"] = w["intent_keywords"]
    if c.review_volatility_90d >= 0.25:
        s += w["review_volatility"]; rationale["review_volatility"] = w["review_volatility"]
    if c.sector_cpc_high:
        s += w["high_cpc_sector"]; rationale["high_cpc_sector"] = w["high_cpc_sector"]
    if c.prior_research_vendor_mentions:
        s += w["prior_research_spend"]; rationale["prior_research_spend"] = w["prior_research_spend"]
    # penalties
    if c.headcount is not None and c.headcount < cfg["headcount"]["min"]:

    price = cfg["pricing_anchor"]
    proofs = []
    if c.locations_18m>=2: proofs.append(f"{c.locations_18m} new locations in 18 months")

app = FastAPI(title="Business Intel Scout", version="0.1.0")

@app.get("/health")
def health():
    h = hashlib.sha256(os.urandom(8)).hexdigest()[:8]
    return {"ok": True, "hash": h}

class ScoreRequest(BaseModel):
    companies: List[CompanyIn]

@app.post("/score", response_model=List[ScoredCompany])
def score(req: ScoreRequest):
    out = []
    for comp in req.companies:
        sc = score_company(comp, CFG)
        out.append(ScoredCompany(company=comp, scores=sc))
    return out

class RankRequest(BaseModel):
    companies: List[CompanyIn]
    top_n: int = Field(default=20, ge=1, le=1000)

@app.post("/rank", response_model=List[ScoredCompany])
def rank(req: RankRequest):
    scored = [ScoredCompany(company=c, scores=score_company(c, CFG)) for c in req.companies]
    ranked = sorted(scored, key=lambda x: x.scores.total, reverse=True)[:req.top_n]
    return ranked

class PitchRequest(BaseModel):
    company: CompanyIn

@app.post("/pitch", response_model=PitchOut)
def pitch(req: PitchRequest):
    sc = score_company(req.company, CFG)
    return make_pitch(req.company, sc, CFG)

def load_config(path: str|None=None) -> Dict[str,Any]:
    global CONFIG_CACHE
    if CONFIG_CACHE: return CONFIG_CACHE
    cfg_path = path or os.getenv("SCOUT_CONFIG", os.path.join(os.path.dirname(__file__), "..", "config.yaml"))
    with open(cfg_path, "r") as f:
        CONFIG_CACHE = yaml.safe_load(f)
    return CONFIG_CACHE

def sigmoid(x: float) -> float:
    return 1/(1+math.exp(-x))

def score_company(c: CompanyIn, cfg: Dict[str,Any]) -> ScoreOut:
    w, p = cfg["weights"], cfg["penalties"]
    rationale = {}
    s = 0.0
    # positives
    if c.headcount is not None and cfg["headcount"]["sweet_min"] <= c.headcount <= cfg["headcount"]["sweet_max"]:
        s += w["size_midmarket"]; rationale["size_midmarket"]=w["size_midmarket"]
    if c.locations_18m >= 2:
        s += w["expansion_locations"]; rationale["expansion_locations"]=w["expansion_locations"]
    if c.funding_months_ago is not None and c.funding_months_ago<=36:
        s += w["funding_recent"]; rationale["funding_recent"]=w["funding_recent"]
    if any(t in set(c.titles) for t in load_config()["titles"]):
        s += w["exec_title_present"]; rationale["exec_title_present"]=w["exec_title_present"]
    if re.search(r"(market entry|competitive analysis|new location|pricing study|research RFP|business intelligence report)", c.intent_text or "", re.I):
        s += w["intent_keywords"]; rationale["intent_keywords"]=w["intent_keywords"]
    if c.review_volatility_90d>=0.25:
        s += w["review_volatility"]; rationale["review_volatility"]=w["review_volatility"]
    if c.sector_cpc_high:
        s += w["high_cpc_sector"]; rationale["high_cpc_sector"]=w["high_cpc_sector"]
    if c.prior_research_vendor_mentions:
        s += w["prior_research_spend"]; rationale["prior_research_spend"]=w["prior_research_spend"]
    # penalties
    if c.headcount is not None and c.headcount < cfg["headcount"]["min"]:
        s += cfg["penalties"]["headcount_too_small"]; rationale["headcount_too_small"]=cfg["penalties"]["headcount_too_small"]
    if c.headcount is not None and c.headcount > cfg["headcount"]["max"]:
        s += cfg["penalties"]["enterprise_slowcycle"]; rationale["enterprise_slowcycle"]=cfg["penalties"]["enterprise_slowcycle"]
    fit = sigmoid(s/4.0)
    urgency = sigmoid((c.locations_18m + (1 if (c.funding_months_ago is not None and c.funding_months_ago<=12) else 0))/2.0)
    total = 0.6*fit + 0.4*urgency
    return ScoreOut(fit=round(fit,3), urgency=round(urgency,3), total=round(total,3), rationale=rationale)

def make_pitch(c: CompanyIn, sc: ScoreOut, cfg: Dict[str,Any]) -> PitchOut:
    price = cfg["pricing_anchor"]
    proofs = []
    if c.locations_18m>=2: proofs.append(f"{c.locations_18m} new locations in 18 months")
    if c.funding_months_ago is not None and c.funding_months_ago<=36: proofs.append("recent funding")
    if re.search(r"competitive|market entry|pricing", c.intent_text or "", re.I): proofs.append("active strategy intent")
    proof_txt = "; ".join(proofs) or "growth activity detected"
    subject = f"Same intelligence, today — {c.name}"
    email = (
        f"{c.name} can skip a ${price['legacy_price']:,} / {price['legacy_timeline_days']}-day study\\n"
        f"SINCOR delivers equivalent decision intel in ~{price['sincor_timeline_hours']} hours for ${price['sincor_price']:,}.\\n"
        f"Why you: {proof_txt}. Headcount ~{c.headcount or 'n/a'}.\\n\\n"
        "Want a 2-page preview by tomorrow?\\n— Reply 'PREVIEW' or hit: action://demo.start\\n"
    )
    dm = (f"Skip ${price['legacy_price']:,}/{price['legacy_timeline_days']}d. "
          f"Same intel in ~{price['sincor_timeline_hours']}h for ${price['sincor_price']:,}. "
          f"{c.name}: {proof_txt}. Preview?")
    return PitchOut(subject=subject, email=email, dm=dm)

# ---- Idempotent outreach store (sqlite) ----
DB_PATH = os.path.join(os.path.dirname(__file__), "..", "scout.db")

def db_init():
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("CREATE TABLE IF NOT EXISTS outreach (hash TEXT PRIMARY KEY, company TEXT, domain TEXT, ts REAL, channel TEXT)")
    cur.execute("CREATE TABLE IF NOT EXISTS outcomes (domain TEXT, won INT, lost INT, created_ts REAL)")
    con.commit(); con.close()

def outreach_hash(domain: str, channel: str) -> str:
    return hashlib.sha256(f"{domain}:{channel}".encode()).hexdigest()

def outreach_seen(domain: str, channel: str, days=14) -> bool:
    import time
    cutoff = time.time() - days*86400
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("SELECT ts FROM outreach WHERE hash=? AND ts>?", (outreach_hash(domain, channel), cutoff))
    row = cur.fetchone()
    con.close()
    return row is not None

def outreach_mark(domain: str, company: str, channel: str):
    import time
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.execute("INSERT OR REPLACE INTO outreach (hash, company, domain, ts, channel) VALUES (?,?,?,?,?)",
                (outreach_hash(domain, channel), company, domain, time.time(), channel))
    con.commit(); con.close()

import os, json, hashlib, time
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, Field
from typing import List, Optional, Dict, Any
from apscheduler.schedulers.background import BackgroundScheduler

from .models import CompanyIn, ScoredCompany, ScoreOut, PitchOut
from .utils import load_config, score_company, make_pitch, db_init, outreach_seen, outreach_mark
from .enrich import enrich
from .connectors.crm import push_lead

CFG = load_config()
db_init()
app = FastAPI(title="Business Intel Scout PRO", version="0.2.0")
scheduler = BackgroundScheduler(daemon=True)
scheduler.start()

@app.get("/health")
def health():
    return {"ok": True, "cfg": bool(CFG), "time": time.time()}

class EnrichRequest(BaseModel):
    domains: List[str]

@app.post("/enrich", response_model=List[CompanyIn])
def enrich_domains(req: EnrichRequest):
    return [enrich(d) for d in req.domains]

class ScoreRequest(BaseModel):
    companies: List[CompanyIn]

@app.post("/score", response_model=List[ScoredCompany])
def score(req: ScoreRequest):
    out = []
    for c in req.companies:
        out.append(ScoredCompany(company=c, scores=score_company(c, CFG)))
    return out

class RankRequest(BaseModel):
    companies: List[CompanyIn]
    top_n: int = Field(default=20, ge=1, le=1000)

@app.post("/rank", response_model=List[ScoredCompany])
def rank(req: RankRequest):
    scored = [ScoredCompany(company=c, scores=score_company(c, CFG)) for c in req.companies]
    scored.sort(key=lambda x: x.scores.total, reverse=True)
    return scored[:req.top_n]

class PitchRequest(BaseModel):
    company: CompanyIn

@app.post("/pitch", response_model=PitchOut)
def pitch(req: PitchRequest):
    sc = score_company(req.company, CFG)
    return make_pitch(req.company, sc, CFG)

class OutreachRequest(BaseModel):
    company: CompanyIn
    channel: str = Field(description="email|linkedin|webhook", default="email")

@app.post("/outreach")
def outreach(req: OutreachRequest):
    c = req.company
    if outreach_seen(c.domain, req.channel):
        return {"ok": False, "reason": "recently contacted", "domain": c.domain, "channel": req.channel}
    sc = score_company(c, CFG)
    p = make_pitch(c, sc, CFG)
    payload = {
        "company": c.dict(),
        "scores": sc.dict(),
        "copy": {"subject": p.subject, "email": p.email, "dm": p.dm},
        "idempotency": hashlib.sha256(f"{c.domain}:{req.channel}".encode()).hexdigest()
    }
    res = push_lead(payload)
    if not res.get("ok"):
        raise HTTPException(status_code=502, detail=f"CRM push failed: {res.get('reason','unknown')}")
    outreach_mark(c.domain, c.name, req.channel)
    return {"ok": True, "pushed": True, "channel": req.channel, "status": res}

# ---- simple tuner endpoint (manual or scheduled) ----
class Outcome(BaseModel):
    domain: str
    won: int = 0
    lost: int = 0

class TuneRequest(BaseModel):
    outcomes: List[Outcome]

@app.post("/tune")
def tune(req: TuneRequest):
    # Very simple weight nudger: if won>lost and score<SQL, boost; if lost>won and score>SQL, dampen.
    cfg = load_config()
    lr = cfg.get("tuning", {}).get("learning_rate", 0.1)
    sql = cfg["thresholds"]["SQL"]
    updated = {}
    for o in req.outcomes:
        c = enrich(o.domain)  # enrich for current signals
        sc = score_company(c, cfg)
        if o.won > o.lost and sc.total < sql/10:  # total in 0..1 vs SQL ~ 7 => normalize
            # boost intent + expansion slightly
            cfg["weights"]["intent_keywords"] += lr*0.1
            cfg["weights"]["expansion_locations"] += lr*0.1
            updated[o.domain] = "boosted"
        elif o.lost > o.won and sc.total > sql/10:
            cfg["weights"]["intent_keywords"] -= lr*0.1
            cfg["weights"]["expansion_locations"] -= lr*0.1
            updated[o.domain] = "dampened"
        else:
            updated[o.domain] = "nochange"
    # save back to config file
    path = os.getenv("SCOUT_CONFIG", os.path.join(os.path.dirname(__file__), "..", "config.yaml"))
    with open(path, "w") as f:
        import yaml; yaml.safe_dump(cfg, f)
    return {"ok": True, "updated": updated, "weights": cfg["weights"]}
""")
